const axios = require('axios');

async function pullCard() {
  const url = 'https://shoob-database-api.vercel.app/api/card';
  
  try {
    const { data } = await axios.get(url, {
      params: {
        random: true,
        key: 'SILENT_TECH_2026'
      }
    });
    
    console.log(data.name, data.tier);
  } catch (err) {
    console.error(err.response.data);
  }
}
bot.py (Python) 
import requests

def pull_card():
    url = "https://shoob-database-api.vercel.app/api/card"
    params = {
        "random": "true",
        "key": "SILENT_TECH_2026"
    }
    
    response = requests.get(url, params=params)
    
    if response.status_code == 200:
        data = response.json()
        print(f"Pulled: {data['name']} [{data['tier']}]")
    else:
        print("Error:", response.text)
